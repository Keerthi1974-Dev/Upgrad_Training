// Importing System namespace for Console class
using System;

// Namespace to group related classes
namespace EmployeeAddressExample
{
    // Address class to store address details
    class Address
    {
        // Properties for address details
        public string Street { get; set; }
        public string City { get; set; }
        public string PinCode { get; set; }

        // Constructor to initialize address
        public Address(string street, string city, string pinCode)
        {
            Street = street;
            City = city;
            PinCode = pinCode;
        }

        // Method to display address
        public void DisplayAddress()
        {
            Console.WriteLine($"Address: {Street}, {City}, {PinCode}");
        }
    }

    // Employee class
    class Employee
    {
        // Employee properties
        public int EmployeeId { get; set; }
        public string Name { get; set; }
        public double Salary { get; set; }

        // HAS-A relationship (Employee has Address)
        public Address EmployeeAddress { get; set; }

        // Constructor to initialize employee
        public Employee(int employeeId, string name, double salary, Address address)
        {
            EmployeeId = employeeId;
            Name = name;
            Salary = salary;
            EmployeeAddress = address;
        }

        // Method to display employee details
        public void DisplayEmployee()
        {
            Console.WriteLine($"Employee ID: {EmployeeId}");
            Console.WriteLine($"Employee Name: {Name}");
            Console.WriteLine($"Salary: INR.{Salary}");

            // Calling Address method
            EmployeeAddress.DisplayAddress();
        }
    }

    // Main class (program execution starts here)
    class Program
    {
        static void Main(string[] args)
        {
            // Creating Address object
            Address addressObj = new Address("15th Line, Ameerpet", "Hyderabad", "500018");

            // Creating Employee object and passing Address
            Employee emp = new Employee(10144, "Narasimha", 75000, addressObj);

            // Display employee details
            emp.DisplayEmployee();

            Console.WriteLine("----------------------------");

            // Changing pincode
            emp.EmployeeAddress.PinCode = "500019";

            // Display updated details
            emp.DisplayEmployee();

            Console.ReadLine();
        }
    }
}
